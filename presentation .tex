% Create a command for the title box
\newcommand{\titlebox}[1]{  
    \begin{tikzpicture}[remember picture,overlay]
        % Create a white box for title, author, and date
        \node[fill=boxcolor, draw=bordercolor, line width=0.1mm, rounded corners=2pt,
              text width=0.7\paperwidth, text centered, inner sep=10mm] 
              at (current page.center) 
              {
                \color{textcolor} % Set text color
                #1 % Content passed to the title box
              };
    \end{tikzpicture}
}
